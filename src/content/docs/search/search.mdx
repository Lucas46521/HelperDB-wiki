async function buscarUsuarios(query) {
    const [porNome, porEmail, porTelefone] = await Promise.all([
        db.search(query, 'nome'),
        db.search(query, 'email'),
        db.search(query, 'telefone')
    ]);

        arr.findIndex(i => i.key === item.key) === index
    );
