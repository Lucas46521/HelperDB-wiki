## Quick Example

```javascript
const { Database } = require('helper.db');

const db = new Database('./data.json');

// Set a value
await db.set('user.123.name', 'John Doe');

// Get a value
const userName = await db.get('user.123.name');
console.log(userName); // 'John Doe'

// Check if a key exists
const exists = await db.has('user.123');
console.log(exists); // true
```
