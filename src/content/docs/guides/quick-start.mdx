# Quick Start

Get up and running with Helper.db in just a few minutes! This guide will walk you through the basics.

## Your First Database

Let's create a simple user management system:

```javascript
const { Database } = require('helper.db');

// Create or connect to a database
const db = new Database('users.db');

async function main() {
  // Create a users table (if it doesn't exist)
  await db.createTable('users', {
    id: 'INTEGER PRIMARY KEY AUTOINCREMENT',
    name: 'TEXT NOT NULL',
    email: 'TEXT UNIQUE',
    age: 'INTEGER',
    created_at: 'DATETIME DEFAULT CURRENT_TIMESTAMP'
  });

  console.log('✅ Database and table created!');
}

main().catch(console.error);
```

## Insert Data

Add some users to your database:

```javascript
async function addUsers() {
  // Insert a single user
  const userId = await db.insert('users', {
    name: 'Alice Johnson',
    email: 'alice@example.com',
    age: 28
  });

  console.log('👤 User created with ID:', userId);

  // Insert multiple users at once
  await db.insertMany('users', [
    { name: 'Bob Smith', email: 'bob@example.com', age: 32 },
    { name: 'Carol Davis', email: 'carol@example.com', age: 25 },
    { name: 'David Wilson', email: 'david@example.com', age: 29 }
  ]);

  console.log('👥 Multiple users added!');
}
```

## Query Data

Retrieve data with simple or complex queries:

```javascript
async function queryUsers() {
  // Get all users
  const allUsers = await db.find('users');
  console.log('All users:', allUsers);

  // Find users by age
  const youngUsers = await db.find('users', { age: { $lt: 30 } });
  console.log('Users under 30:', youngUsers);

  // Find a specific user
  const alice = await db.findOne('users', { email: 'alice@example.com' });
  console.log('Alice:', alice);

  // Advanced query with sorting
  const sortedUsers = await db.query()
    .from('users')
    .where('age', '>', 25)
    .orderBy('name', 'asc')
    .limit(5)
    .execute();
  
  console.log('Sorted results:', sortedUsers);
}
```

## Update Data

Modify existing records:

```javascript
async function updateUsers() {
  // Update a single user
  await db.update('users', 
    { email: 'alice@example.com' }, // condition
    { age: 29 }                     // update data
  );

  // Update multiple users
  await db.updateMany('users',
    { age: { $lt: 30 } },           // condition
    { status: 'young' }             // update data
  );

  console.log('✏️ Users updated!');
}
```

## Delete Data

Remove records when needed:

```javascript
async function deleteUsers() {
  // Delete a specific user
  await db.delete('users', { email: 'bob@example.com' });

  // Delete multiple users
  await db.deleteMany('users', { age: { $gt: 35 } });

  console.log('🗑️ Users deleted!');
}
```

## Complete Example

Here's a complete working example:

```javascript
const { Database } = require('helper.db');

async function userManagementExample() {
  const db = new Database('example.db');

  try {
    // Setup
    await db.createTable('users', {
      id: 'INTEGER PRIMARY KEY AUTOINCREMENT',
      name: 'TEXT NOT NULL',
      email: 'TEXT UNIQUE',
      age: 'INTEGER',
      status: 'TEXT DEFAULT "active"',
      created_at: 'DATETIME DEFAULT CURRENT_TIMESTAMP'
    });

    // Add users
    await db.insertMany('users', [
      { name: 'Emma Thompson', email: 'emma@example.com', age: 27 },
      { name: 'James Brown', email: 'james@example.com', age: 34 },
      { name: 'Sofia Martinez', email: 'sofia@example.com', age: 22 }
    ]);

    // Query users
    const activeUsers = await db.find('users', { status: 'active' });
    console.log(`Found ${activeUsers.length} active users`);

    // Update user
    await db.update('users', 
      { name: 'Emma Thompson' }, 
      { age: 28, status: 'premium' }
    );

    // Advanced query
    const premiumUsers = await db.query()
      .from('users')
      .where('status', '=', 'premium')
      .select(['name', 'email', 'age'])
      .execute();

    console.log('Premium users:', premiumUsers);

  } catch (error) {
    console.error('Error:', error);
  } finally {
    await db.close();
  }
}

userManagementExample();
```

## Query Builder

Helper.db includes a powerful query builder for complex operations:

```javascript
// Complex query with joins and conditions
const results = await db.query()
  .from('users')
  .join('orders', 'users.id', '=', 'orders.user_id')
  .where('users.age', '>', 18)
  .and('orders.status', '=', 'completed')
  .groupBy('users.id')
  .having('COUNT(orders.id)', '>', 5)
  .orderBy('users.created_at', 'desc')
  .limit(10)
  .execute();
```

## Backup and Restore

Protect your data with built-in backup functionality:

```javascript
const { Backup } = require('helper.db');

async function backupExample() {
  const backup = new Backup(db);

  // Create a backup
  await backup.create('backup-' + Date.now() + '.db');
  console.log('📥 Backup created!');

  // List all backups
  const backups = await backup.list();
  console.log('Available backups:', backups);

  // Restore from backup (if needed)
  // await backup.restore('backup-1234567890.db');
}
```
